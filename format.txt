boolean - 1 byte
word - 2 bytes, big endian
dword - 4 bytes, big endian
long - 8 bytes, big edian
string - UTF-8

[byte] * 4 = "TCSG"
[word] = version
[word][string] - author
[word][string] - description
[long] - last save UNIX time
[word] - blocks count
[byte] = block type 
	1 - colors
	2 - fragment
For 1 - colors
[word] - colors count
{
	[dword] - color
	[word][string] - name
	...
}
For 2 - fragments
[word] - fragments count
{
	[word][string] - name
	[word] - template width
	[word] - template height
	[word] - primitives count
	{
		[word] - primitive type
			1 - Point
				[word] - x
				[word] - y
				[dword] - color
				[boolean] - proportional flag
			2 - Line
				[word] - x1
				[word] - y1
				[word] - x2
				[word] - y2
				[dword] - color
				[boolean] - proportional flag
			3 - Rect
				[word] - x
				[word] - y
				[word] - width
				[word] - height
				[dword] - color
				[boolean] - proportional flag
				[boolean] - fill flag
			4 - Gradient
				[word] - x
				[word] - y
				[word] - width
				[word] - height
				[dword] - color start
				[dword] - color end
				[boolean] - proportional flag
				[boolean] - fill flag
				[boolean] - vertical flag
		...
	}
	...
}
